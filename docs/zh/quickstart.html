安装

将 goscapy 添加到你的项目中：

bash 
go get github.com/smallnest/goscapy

导入所需的包：

go 
import (
    "fmt"
    "time"

    "github.com/smallnest/goscapy/pkg/goscapy"
    "github.com/smallnest/goscapy/pkg/packet"
    "github.com/smallnest/goscapy/pkg/sendrecv"
    "github.com/smallnest/goscapy/pkg/sniff"
)
📝 注意
原始套接字操作需要 root/管理员权限。在 Linux 或 macOS 上请使用 sudo 运行程序。

构建数据包（Builder API）

Builder API 提供流畅的方法链式调用，实现类型安全的显式数据包构建。 每个协议层都有自己的 builder，通过方法设置字段。使用 .Over() 将各层串联起来，最后调用 .Build() 完成构建。

Ethernet + IP + ICMP

go 
// 从零构建 ICMP Echo Request 数据包
pkt, err := goscapy.NewEthernet().
    SrcMAC("aa:bb:cc:dd:ee:ff").
    DstMAC("ff:ff:ff:ff:ff:ff").
    Over(goscapy.NewIP().
        SrcIP("192.168.1.1").
        DstIP("8.8.8.8").
        TTL(64)).
    Over(goscapy.NewICMP().
        Type(8).   // Echo Request
        Code(0).
        ID(1).
        Seq(1)).
    Build()
if err != nil {
    log.Fatal(err)
}
fmt.Printf("构建了 %d 字节\n", len(pkt))

Ethernet + IP + TCP

go 
// 构建 TCP SYN 数据包
pkt, err := goscapy.NewEthernet().
    SrcMAC("aa:bb:cc:dd:ee:ff").
    DstMAC("11:22:33:44:55:66").
    Over(goscapy.NewIP().
        SrcIP("192.168.1.100").
        DstIP("93.184.216.34")).
    Over(goscapy.NewTCP().
        SrcPort(12345).
        DstPort(80).
        Flags(0x02).  // SYN
        Seq(1000).
        Window(65535)).
    Build()

Ethernet + IP + UDP

go 
// 构建 UDP 数据包
pkt, err := goscapy.NewEthernet().
    SrcMAC("aa:bb:cc:dd:ee:ff").
    DstMAC("11:22:33:44:55:66").
    Over(goscapy.NewIP().
        SrcIP("192.168.1.100").
        DstIP("8.8.8.8")).
    Over(goscapy.NewUDP().
        SrcPort(54321).
        DstPort(53)).
    Build()
💡 提示
IP、TCP、UDP、ICMP 的校验和在 .Build() 时自动计算，无需手动设置。

构建数据包（快捷函数）

快捷函数只需一个函数调用即可构建常用的协议栈。它们使用合理的默认值 （如 srcMAC="00:00:00:00:00:00"、srcIP="0.0.0.0"）， 你只需指定关键参数。

go 
// ICMP Echo Request — 一行搞定
pkt, err := goscapy.EtherIPICMP("ff:ff:ff:ff:ff:ff", "8.8.8.8", 8, 0)

// TCP SYN 数据包
pkt, err := goscapy.EtherIPTCP(
    "aa:bb:cc:dd:ee:ff", "11:22:33:44:55:66",
    "192.168.1.100", "93.184.216.34",
    12345, 80, 0x02,
)

// UDP 数据包
pkt, err := goscapy.EtherIPUDP(
    "aa:bb:cc:dd:ee:ff", "11:22:33:44:55:66",
    "192.168.1.100", "8.8.8.8",
    54321, 53,
)

// ARP 请求
pkt, err := goscapy.EtherARP(
    "aa:bb:cc:dd:ee:ff", "ff:ff:ff:ff:ff:ff",
    "192.168.1.100", "192.168.1.1",
    1, // ARP Request
)

// 纯 IP 数据包（无以太网头）
pkt, err := goscapy.IPICMP("192.168.1.100", "8.8.8.8", 8, 0)
pkt, err := goscapy.IPTCP("192.168.1.100", "93.184.216.34", 12345, 80, 0x02)
pkt, err := goscapy.IPUDP("192.168.1.100", "8.8.8.8", 54321, 53)

解析数据包

将原始字节解析回结构化的 Packet 对象。goscapy 自动检测每个层的协议 并提取所有字段。完整以太网帧使用 DissectEthernet，纯 IP 数据使用 DissectIP。

go 
// 先构建一个数据包用于解析
raw, _ := goscapy.EtherIPICMP("ff:ff:ff:ff:ff:ff", "8.8.8.8", 8, 0)

// 解析回分层结构
pkt, err := packet.Dissect(raw, packet.DissectEthernet)
if err != nil {
    log.Fatal(err)
}

// 打印层栈
fmt.Println(pkt.String()) // "Ethernet / IP / ICMP"

// 访问各个层
eth := packet.Ethernet(pkt)
fmt.Printf("源MAC: %s\n", eth.SrcMAC())
fmt.Printf("目标MAC: %s\n", eth.DstMAC())

ip := packet.IP(pkt)
fmt.Printf("源IP: %s  目标IP: %s  TTL: %d\n",
    ip.SrcIP(), ip.DstIP(), ip.TTL())
💡 提示
当原始字节从 IP 头开始（无以太网帧）时，使用 packet.DissectIP 代替 packet.DissectEthernet。

发送数据包

goscapy 支持在两个层级发送数据包：

  • L3（Send） — 在 IP 层级发送。操作系统自动添加以太网头。适用于 IP+传输层的数据包。
  • L2（Sendp） — 发送完整的以太网帧。你控制整个数据包。
go 
// 构建 ICMP Echo Request
raw, _ := goscapy.EtherIPICMP("ff:ff:ff:ff:ff:ff", "8.8.8.8", 8, 0)
pkt, _ := packet.Dissect(raw, packet.DissectEthernet)

// L3 发送 — 操作系统处理以太网帧
err := sendrecv.Send(pkt, "eth0")

// L2 发送 — 你提供完整的帧
err = sendrecv.Sendp(pkt, "eth0")

发送并接收响应

使用 SendRecv 发送数据包并收集响应，或使用 SendRecv1 只获取第一个响应：

go 
// 发送 ICMP Echo 并等待回复
raw, _ := goscapy.EtherIPICMP("ff:ff:ff:ff:ff:ff", "8.8.8.8", 8, 0)
pkt, _ := packet.Dissect(raw, packet.DissectEthernet)

// 在 3 秒内获取所有响应
sent, responses, err := sendrecv.SendRecv(pkt, "eth0", 3*time.Second)
fmt.Printf("收到 %d 个响应\n", len(responses))

// 或只获取第一个响应
sent, reply, err := sendrecv.SendRecv1(pkt, "eth0", 3*time.Second)
if reply != nil {
    fmt.Println("回复:", reply.String())
}
📝 注意
SendRecv 在发送之前打开接收器，以避免错过快速响应（如在回环接口上）。

嗅探数据包

使用 回调处理器 或 通道 捕获实时流量。 两者都支持 BPF 过滤表达式（需要系统 PATH 中有 tcpdump 来编译过滤规则）。

回调式嗅探

go 
// 嗅探 ICMP 数据包，持续 10 秒
err := sniff.Sniff(sniff.SniffConfig{
    Iface:   "eth0",
    Filter:  "icmp",
    Timeout: 10 * time.Second,
}, func(pkt *packet.Packet) bool {
    fmt.Println(pkt.String())
    return true // 继续嗅探
})

通道式嗅探

go 
// 精确捕获 5 个 TCP 数据包
ch, stop := sniff.SniffChan(sniff.SniffConfig{
    Iface:  "eth0",
    Filter: "tcp",
    Count:  5,
})
defer stop()

for pkt := range ch {
    fmt.Println(pkt.String())
}
📝 注意
BPF 过滤器在内核层面编译，确保最高性能。使用 sniff.CompileFilter("tcp port 80") 可以预编译过滤字符串。
