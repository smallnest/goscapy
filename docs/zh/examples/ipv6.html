IPv6 + ICMPv6 Echo Request

构建 IPv6 ping 数据包。ICMPv6 Echo 使用双层结构：ICMPv6 基础头（Type + Code + Checksum）和 Echo 子层（ID + Seq）。IPv6 的 TC（流量类别）、FL（流标签）和 HLim（跳数限制）字段控制 QoS 和路由行为。伪头部校验和自动计算。

🔨 Builder API

go
import (
    "github.com/smallnest/goscapy/pkg/goscapy"
    "github.com/smallnest/goscapy/pkg/layers"
)

raw, err := goscapy.NewIPv6().
    SrcIP("fd00::1").
    DstIP("fd00::2").
    TC(0).                // 流量类别 (0–255)
    FL(0).                // 流标签 (20 位, 0–1048575)
    HLim(64).              // 跳数限制
    Over(goscapy.NewICMPv6().
        Type(128).
        Code(0)).
    Over(&goscapy.RawBuilder{Layer: layers.NewICMPv6Echo(0x1234, 1)}).
    Build()

⚡ 快捷函数

go
// IPv6ICMPv6Echo(srcIP, dstIP string, id, seq uint16)
raw, err := goscapy.IPv6ICMPv6Echo("fd00::1", "fd00::2", 0x1234, 1)

🔬 解析

go
pkt, err := packet.Dissect(raw, packet.DissectIPv6)
fmt.Println(pkt.String()) // "IPv6 / ICMPv6 / ICMPv6 Echo"

ip6 := pkt.GetLayer("IPv6")
srcIP, _ := ip6.Get("srcIP")
dstIP, _ := ip6.Get("dstIP")
hlim, _  := ip6.Get("hopLimit")
fmt.Printf("%s → %s  HLim=%d\n", srcIP, dstIP, hlim)

IPv6 + TCP SYN

构建基于 IPv6 的 TCP SYN 数据包。Next Header 字段（NH）根据上层协议自动设置（6 = TCP）。TCP 校验和计算使用 IPv6 伪头部。

🔨 Builder API

go
raw, err := goscapy.NewIPv6().
    SrcIP("fd00::1").
    DstIP("fd00::2").
    HLim(64).
    Over(goscapy.NewTCP().
        SrcPort(12345).
        DstPort(80).
        Flags(layers.TCPSyn).
        Seq(1000)).
    Build()

🔬 解析

go
pkt, err := packet.Dissect(raw, packet.DissectIPv6)
fmt.Println(pkt.String()) // "IPv6 / TCP"

tcp := pkt.GetLayer("TCP")
srcPort, _ := tcp.Get("srcPort")
dstPort, _ := tcp.Get("dstPort")
fmt.Printf("端口 %d → %d  SYN\n", srcPort, dstPort)

IPv6 + UDP

构建基于 IPv6 的 UDP 数据报。Next Header 自动设置为 17 (UDP)。这是基于 IPv6 的 DNS、DHCPv6 和 VXLAN 的基础。

NDP Router Solicitation

构建 NDP 路由器请求 — IPv6 主机用于在本地链路上发现路由器。消息使用 ICMPv6 类型 133，可包含 Source Link-Layer Address 选项以便路由器直接响应。

🔨 Builder API

go
import "github.com/smallnest/goscapy/pkg/layers/ndp"

// 构建 NDP 选项：源链路层地址
ndpOpts := ndp.BuildNDPOptions([]ndp.NDPOption{
    ndp.BuildSourceLinkLayerOption(net.HardwareAddr{0x00, 0x11, 0x22, 0x33, 0x44, 0x55}),
})

raw, err := goscapy.NewIPv6().
    SrcIP("fe80::211:22ff:fe33:4455").
    DstIP("ff02::2").            // 所有路由器组播
    HLim(255).
    Over(goscapy.NewICMPv6().
        Type(ndp.NDPRouterSolicitation).  // 133
        Code(0)).
    Over(&goscapy.RawBuilder{Layer: layers.NewNDPRouterSolicitation()}).
    Build()

🔬 解析

go
pkt, err := packet.Dissect(raw, packet.DissectIPv6)
fmt.Println(pkt.String()) // "IPv6 / ICMPv6 / NDP Router Solicitation"

icmpLayer := pkt.GetLayer("ICMPv6")
msgType, _ := icmpLayer.Get("type")
fmt.Printf("ICMPv6 Type=%d (Router Solicitation)\n", msgType)

NDP Neighbor Advertisement

构建 NDP 邻居通告 — 用于响应邻居请求或通告链路层地址变更。Router、Solicited 和 Override 标志控制接收方的邻居缓存行为。

🔨 Builder API

go
// 构建 NDP 选项：目标链路层地址
ndpOpts := ndp.BuildNDPOptions([]ndp.NDPOption{
    ndp.BuildTargetLinkLayerOption(net.HardwareAddr{0x00, 0xaa, 0xbb, 0xcc, 0xdd, 0xee}),
})

raw, err := goscapy.NewIPv6().
    SrcIP("fd00::1").
    DstIP("fd00::2").
    HLim(255).
    Over(goscapy.NewICMPv6().
        Type(ndp.NDPNeighborAdvertisement). // 136
        Code(0)).
    Over(&goscapy.RawBuilder{Layer: layers.NewNDPNeighborAdvertisement()}).
    Build()

🔬 解析

go
pkt, err := packet.Dissect(raw, packet.DissectIPv6)
fmt.Println(pkt.String()) // "IPv6 / ICMPv6 / NDP Neighbor Advertisement"

icmpLayer := pkt.GetLayer("ICMPv6")
msgType, _ := icmpLayer.Get("type")
fmt.Printf("ICMPv6 Type=%d (Neighbor Advertisement)\n", msgType)

ndpLayer := pkt.GetLayer("NDP Neighbor Advertisement")
if ndpLayer != nil {
    flags, _ := ndpLayer.Get("flags")
    fmt.Printf("Flags=0x%02x  Router=%v  Solicited=%v\n",
        flags,
        flags.(uint8)&ndp.NDPNARouter != 0,
        flags.(uint8)&ndp.NDPNASolicited != 0)
}

IPv6 扩展头链

构建包含扩展头链的 IPv6 数据包：Hop-by-Hop Options → Fragment → TCP 载荷。扩展头通过 Over() 逐层堆叠，Next Header 字段在每一跳自动设置。Fragment 头携带偏移量和 More Fragments (M) 标志用于 IPv6 分片。

🔨 Builder API

go
import "github.com/smallnest/goscapy/pkg/layers/ipv6"

// IPv6 带 Hop-by-Hop → Fragment → TCP 扩展头链
raw, err := goscapy.NewIPv6().
    SrcIP("fd00::1").
    DstIP("fd00::2").
    HLim(64).
    // Next Header = 0 (Hop-by-Hop) 自动设置
    Over(&goscapy.RawBuilder{Layer: ipv6.NewIPv6HopByHop()}).
    // Next Header = 44 (Fragment) 自动设置
    Over(&goscapy.RawBuilder{Layer: ipv6.NewIPv6Fragment()}).
    // 最终 Next Header = 6 (TCP)
    Over(goscapy.NewTCP().
        SrcPort(12345).
        DstPort(80).
        Flags(layers.TCPSyn)).
    Build()

🔬 解析

go
pkt, err := packet.Dissect(raw, packet.DissectIPv6)
fmt.Println(pkt.String())
// "IPv6 / IPv6 Hop-by-Hop / IPv6 Fragment / TCP"

// 遍历扩展头链
for _, layer := range pkt.Layers() {
    fmt.Printf("层: %s\n", layer.Proto())
}

// 访问 Fragment 头字段
fragLayer := pkt.GetLayer("IPv6 Fragment")
if fragLayer != nil {
    fragData, _ := fragLayer.Get("fragment")
    offset := ipv6.IPv6FragmentOffset(fragData.(uint16))
    more    := ipv6.IPv6FragmentMore(fragData.(uint16))
    fmt.Printf("分片: offset=%d  more=%v\n", offset, more)
}
注意：IPv6 扩展头通过 Next Header 字段链接。每个扩展头指示下一个头的类型。最后一个扩展头指向上层协议（TCP/UDP/ICMPv6）。Builder 根据 Over() 链自动设置 Next Header。
