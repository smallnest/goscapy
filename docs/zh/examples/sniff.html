基本嗅探 — 捕获所有数据包

使用回调 API 在指定接口上捕获所有数据包。每个数据包都会调用处理函数 —— 返回 true 继续，返回 false 停止。

🔨 示例

go
import (
    "fmt"
    "time"

    "github.com/smallnest/goscapy/pkg/packet"
    "github.com/smallnest/goscapy/pkg/sniff"
)

err := sniff.Sniff(sniff.SniffConfig{
    Iface:   "eth0",
    Timeout: 10 * time.Second,
}, func(pkt *packet.Packet) bool {
    // 打印每个数据包的层信息摘要
    fmt.Printf("[%s] %s\n", time.Now().Format("15:04:05"), pkt.String())

    // 访问特定协议层
    if ip := packet.IP(pkt); ip != nil {
        fmt.Printf("  %s → %s\n", ip.SrcIP(), ip.DstIP())
    }

    return true // true = 继续捕获
})
if err != nil {
    fmt.Fatal(err)
}
📝 注意
Sniff() 会阻塞，直到超时、达到数量限制或处理函数返回 false。 内部每次读取超时为 1 秒以保证响应性 —— ErrTimeout 会被内部处理并自动重试。

带 BPF 过滤器嗅探

使用 Filter 字段只捕获匹配的数据包。过滤表达式使用与 tcpdump 相同的语法，编译为内核级 BPF 指令以获得最高效率。

🔨 过滤字符串

go
import (
    "fmt"
    "time"

    "github.com/smallnest/goscapy/pkg/packet"
    "github.com/smallnest/goscapy/pkg/sniff"
)

// 只捕获 ICMP 数据包（ping）
err := sniff.Sniff(sniff.SniffConfig{
    Iface:   sendrecv.LoopbackName(),
    Filter:  "icmp",
    Timeout: 10 * time.Second,
}, func(pkt *packet.Packet) bool {
    icmp := packet.ICMP(pkt)
    if icmp != nil {
        fmt.Printf("ICMP type=%d code=%d\n", icmp.Type(), icmp.Code())
    }
    return true
})

🔧 预编译指令

go
// 在启动时预编译 —— 避免运行时依赖 tcpdump
instructions, _ := sniff.CompileFilter("tcp port 80")

// 使用预编译指令（优先级高于 Filter 字段）
err := sniff.Sniff(sniff.SniffConfig{
    Iface:        "eth0",
    Instructions: instructions, // 设置 Instructions 后 Filter 字段被忽略
    Timeout:      10 * time.Second,
}, func(pkt *packet.Packet) bool {
    tcp := packet.TCP(pkt)
    if tcp != nil {
        fmt.Printf("HTTP: %d → %d\n", tcp.SrcPort(), tcp.DstPort())
    }
    return true
})
📝 注意
优先级： 同时设置 Instructions 和 Filter 时，Instructions 优先。 可以在启动时编译过滤器并重复使用，避免运行时依赖 tcpdump。

嗅探 — 限制捕获数量

使用 Count 字段捕获固定数量的数据包后自动停止。可结合 Filter 捕获 N 个匹配特定模式的数据包。

🔨 示例

go
// 精确捕获 5 个 ICMP 数据包后停止
err := sniff.Sniff(sniff.SniffConfig{
    Iface:  sendrecv.LoopbackName(),
    Filter: "icmp",
    Count:  5,
}, func(pkt *packet.Packet) bool {
    ip := packet.IP(pkt)
    icmp := packet.ICMP(pkt)
    fmt.Printf("%s → %s  ICMP type=%d\n",
        ip.SrcIP(), ip.DstIP(), icmp.Type())
    return true
})
💡 提示
设置 Count: 0（默认值）表示无限制捕获。 可以同时设置 Count 和 Timeout —— 嗅探在任一条件满足时停止。

嗅探 — 带超时

使用 Timeout 字段限制嗅探持续时间。设置 Timeout: 0（默认值）表示无限时长。

🔨 示例

go
// 捕获 TCP 流量，持续 30 秒
err := sniff.Sniff(sniff.SniffConfig{
    Iface:   "eth0",
    Filter:  "tcp",
    Timeout: 30 * time.Second,
}, func(pkt *packet.Packet) bool {
    tcp := packet.TCP(pkt)
    fmt.Printf("TCP %d → %d flags=0x%02x\n",
        tcp.SrcPort(), tcp.DstPort(), tcp.Flags())
    return true
})
fmt.Println("嗅探完成。")

嗅探 — 从处理函数中停止

从处理函数返回 false 可立即停止嗅探。当你找到目标数据包时很有用。

🔨 示例

go
import (
    "fmt"
    "github.com/smallnest/goscapy/pkg/layers"
    "github.com/smallnest/goscapy/pkg/packet"
    "github.com/smallnest/goscapy/pkg/sniff"
)

// 嗅探直到看到来自特定主机的 ICMP 回显回复
err := sniff.Sniff(sniff.SniffConfig{
    Iface:  sendrecv.LoopbackName(),
    Filter: "icmp",
    Count:  100, // 安全限制
}, func(pkt *packet.Packet) bool {
    ip := packet.IP(pkt)
    icmp := packet.ICMP(pkt)

    // 检查是否为来自 127.0.0.1 的 ICMP 回显回复
    if icmp != nil && icmp.Type() == layers.ICMPEchoReply {
        fmt.Printf("找到来自 %s 的回复！\n", ip.SrcIP())
        return false // 立即停止嗅探
    }

    fmt.Println("等待回复...")
    return true // 继续
})

SniffChan — 通道式嗅探

SniffChan() 返回一个数据包传输通道和一个停止函数。在与 Go 的 select 和通道模式集成时，比回调更方便。

🔨 示例

go
import (
    "fmt"
    "time"

    "github.com/smallnest/goscapy/pkg/sniff"
)

// 开始嗅探 —— 返回通道和停止函数
ch, stop := sniff.SniffChan(sniff.SniffConfig{
    Iface:   "eth0",
    Filter:  "tcp port 80",
    Count:   10,
    Timeout: 30 * time.Second,
})
defer stop() // 始终调用 stop 以释放资源

// 遍历通道接收数据包
for pkt := range ch {
    fmt.Println(pkt.String())
}
fmt.Println("通道关闭 —— 嗅探完成。")

🔧 使用 Select 提前停止

go
ch, stop := sniff.SniffChan(sniff.SniffConfig{
    Iface:  "eth0",
    Filter: "tcp port 80",
})

// 使用 select 自定义停止条件
timeout := time.After(5 * time.Second)
for {
    select {
    case pkt, ok := <-ch:
        if !ok {
            fmt.Println("完成。")
            return
        }
        fmt.Println(pkt.String())
    case <-timeout:
        fmt.Println("自定义超时。")
        stop()
    }
}

SniffChan — Worker Pool 并发处理

将 SniffChan() 与 worker pool 结合，实现数据包的并发处理。当数据包分析比较耗时（如协议解码、负载检查）时非常有用。

🔨 示例

go
import (
    "fmt"
    "sync"
    "time"

    "github.com/smallnest/goscapy/pkg/packet"
    "github.com/smallnest/goscapy/pkg/sniff"
)

ch, stop := sniff.SniffChan(sniff.SniffConfig{
    Iface:   "eth0",
    Filter:  "tcp",
    Timeout: 30 * time.Second,
})
defer stop()

// 启动 4 个 worker 协程
var wg sync.WaitGroup
for i := 0; i < 4; i++ {
    wg.Add(1)
    go func(id int) {
        defer wg.Done()
        for pkt := range ch {
            // 每个 worker 独立处理数据包
            tcp := packet.TCP(pkt)
            ip := packet.IP(pkt)
            if tcp != nil && ip != nil {
                fmt.Printf("[worker %d] %s:%d → %s:%d\n",
                    id, ip.SrcIP(), tcp.SrcPort(), ip.DstIP(), tcp.DstPort())
            }
        }
    }(i)
}

// 等待所有 worker 完成（SniffChan 关闭通道）
wg.Wait()
fmt.Println("所有 worker 已完成。")
📝 注意
SniffChan() 创建的通道缓冲区大小为 64。 这意味着捕获协程可以缓冲最多 64 个数据包而不会阻塞。 如果 worker 处理速度慢于捕获速度，数据包会先缓冲； 缓冲区满时，捕获协程会阻塞直到 worker 消费一个数据包。

Sniff 与 SniffChan — 如何选择

两种 API 使用相同的底层捕获机制，根据处理模式选择即可。

特性Sniff() — 回调SniffChan() — 通道
处理模型同步回调异步通道 + 协程
停止方式处理函数返回 false调用 stop() 函数
并发单协程（顺序处理）多 worker 通过通道并发
缓冲无缓冲（内联处理）64 个数据包缓冲
适用场景简单日志、计数、快速脚本重型处理、worker pool、流水线架构
错误处理直接返回 error错误导致通道关闭
💡 提示
简单脚本用 Sniff()；需要并发处理、通过 select 自定义停止条件、 或与其他 Go 通道系统集成时用 SniffChan()。

组合使用 — Count + Timeout + Filter

所有配置选项可以组合使用。嗅探在任一条件满足时停止：达到数量限制、超时或处理函数返回 false。

🔨 完整示例

go
package main

import (
    "fmt"
    "os"
    "time"

    "github.com/smallnest/goscapy/pkg/layers"
    "github.com/smallnest/goscapy/pkg/packet"
    "github.com/smallnest/goscapy/pkg/sendrecv"
    "github.com/smallnest/goscapy/pkg/sniff"
)

func main() {
    // 在启动时预编译过滤器
    instructions, err := sniff.CompileFilter("tcp port 80 or tcp port 443")
    if err != nil {
        fmt.Fprintf(os.Stderr, "编译过滤器: %v\n", err)
        os.Exit(1)
    }

    // 捕获最多 50 个 HTTP/HTTPS 数据包，持续 1 分钟
    captured := 0
    err = sniff.Sniff(sniff.SniffConfig{
        Iface:        sendrecv.LoopbackName(),
        Instructions: instructions,
        Count:        50,
        Timeout:      60 * time.Second,
    }, func(pkt *packet.Packet) bool {
        captured++
        ip := packet.IP(pkt)
        tcp := packet.TCP(pkt)

        // 判断流量方向
        dir := "→"
        if tcp.DstPort() == 80 || tcp.DstPort() == 443 {
            dir = "→ 请求"
        } else {
            dir = "← 响应"
        }

        // 检查 SYN 标志
        flags := tcp.Flags()
        if flags&layers.TCPSyn != 0 && flags&layers.TCPAck == 0 {
            fmt.Printf("[%d] %s:%d %s %s:%d SYN\n",
                captured, ip.SrcIP(), tcp.SrcPort(), dir, ip.DstIP(), tcp.DstPort())
        } else {
            fmt.Printf("[%d] %s:%d %s %s:%d\n",
                captured, ip.SrcIP(), tcp.SrcPort(), dir, ip.DstIP(), tcp.DstPort())
        }

        return true
    })
    if err != nil {
        fmt.Fatal(err)
    }
    fmt.Printf("共捕获 %d 个数据包。\n", captured)
}
配置字段默认值说明
Iface（必填）要嗅探的网络接口
Filter""BPF 过滤表达式（需要 tcpdump）
Instructionsnil预编译 BPF 指令（优先级高于 Filter）
Count0最大捕获数量（0 = 无限制）
Timeout0最长持续时间（0 = 无超时）
