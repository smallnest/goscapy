DNS 查询 — A 记录查询

构建 DNS A 记录查询数据包。标准 DNS 查询封装在 Ethernet + IP + UDP 端口 53 中。DNSQuestion 结构体包含域名、查询类型和类别。EncodeName() 将点分格式转换为 DNS 线格式标签。

🔨 Builder API

go
import (
    "github.com/smallnest/goscapy/pkg/goscapy"
    "github.com/smallnest/goscapy/pkg/layers/dns"
)

// 定义 DNS 查询问题
questions := []dns.DNSQuestion{
    {Name: "example.com", Type: dns.QtypeA, Class: dns.QclassIN},
}

// 构建 Ethernet + IP + UDP + DNS 查询
raw, err := goscapy.NewEthernet().
    SrcMAC("00:11:22:33:44:55").
    DstMAC("ff:ff:ff:ff:ff:ff").
    Over(goscapy.NewIP().
        SrcIP("192.168.1.1").
        DstIP("8.8.8.8")).
    Over(goscapy.NewUDP().
        SrcPort(12345).
        DstPort(53)).
    Over(goscapy.NewDNS().
        ID(0x1234).
        Flags(0x0100).         // RD=1 (期望递归)
        Questions(questions)).
    Build()

⚡ 快捷函数

go
// EtherIPUDPDNS(srcMAC, dstMAC, srcIP, dstIP string, dnsPort uint16, questions []dns.DNSQuestion)
questions := []dns.DNSQuestion{
    {Name: "example.com", Type: dns.QtypeA, Class: dns.QclassIN},
}
raw, err := goscapy.EtherIPUDPDNS(
    "00:11:22:33:44:55", "ff:ff:ff:ff:ff:ff",
    "192.168.1.1", "8.8.8.8",
    53,
    questions,
)

🔬 解析

go
import "github.com/smallnest/goscapy/pkg/packet"

pkt, err := packet.Dissect(raw, packet.DissectEthernet)
fmt.Println(pkt.String()) // "Ethernet / IP / UDP / DNS"

// 提取 DNS 查询问题
dnsLayer := pkt.GetLayer("DNS")
questions, err := dns.GetQuestions(dnsLayer)
for _, q := range questions {
    fmt.Printf("查询: %s  Type=%d  Class=%d\n", q.Name, q.Type, q.Class)
}

DNS 响应 — A 记录应答

构建包含 A 记录应答的 DNS 响应。响应设置了 QR 标志位（flags 0x8180 = QR + RD + RA），同时包含查询部分和应答部分。使用 BuildARData() 构建 A 记录的 RDATA。

🔨 Builder API

go
questions := []dns.DNSQuestion{
    {Name: "example.com", Type: dns.QtypeA, Class: dns.QclassIN},
}
answers := []dns.DNSRR{
    {
        Name:     "example.com",
        Type:     dns.QtypeA,
        Class:    dns.QclassIN,
        TTL:      300,
        RDLength: 4,
        RData:    dns.BuildARData("93.184.216.34"),
    },
}

// 构建 DNS 响应数据（查询 + 应答）
dnsData := dns.BuildDNSMessage(questions, answers, nil, nil)

raw, err := goscapy.NewEthernet().
    SrcMAC("00:aa:bb:cc:dd:ee").
    DstMAC("00:11:22:33:44:55").
    Over(goscapy.NewIP().
        SrcIP("8.8.8.8").
        DstIP("192.168.1.1")).
    Over(goscapy.NewUDP().
        SrcPort(53).
        DstPort(12345)).
    Over(goscapy.NewDNS().
        ID(0x1234).
        Flags(0x8180).        // QR=1, RD=1, RA=1
        Data(dnsData)).
    Build()

🔬 解析

go
pkt, err := packet.Dissect(raw, packet.DissectEthernet)

dnsLayer := pkt.GetLayer("DNS")

// 解析查询问题
questions, _ := dns.GetQuestions(dnsLayer)

// 解析应答资源记录
answers, _ := dns.GetAnswers(dnsLayer)
for _, rr := range answers {
    fmt.Printf("应答: %s  Type=%d  TTL=%d", rr.Name, rr.Type, rr.TTL)
    if rr.Type == dns.QtypeA {
        fmt.Printf("  IP=%s", dns.ParseARData(rr.RData))
    }
    fmt.Println()
}
提示：DNS 名称编码使用标签压缩 — 响应中重复的域名通过指针引用之前的出现位置（最高 2 位 = 0xC0）。DecodeName() 会自动处理标签压缩。

DHCP Discover

构建 DHCP Discover 数据包 — 客户端发送的第一条消息，用于在网络中寻找 DHCP 服务器。使用广播地址（255.255.255.255）和 UDP 端口 68（客户端）→ 67（服务器）。MessageType TLV 选项（53）设置为 DHCPDISCOVER。

🔨 Builder API

go
import (
    "github.com/smallnest/goscapy/pkg/goscapy"
    "github.com/smallnest/goscapy/pkg/layers/dhcp"
    "github.com/smallnest/goscapy/pkg/fields"
)

// 构建 DHCP 选项：MessageType + 参数请求列表 + End
opts := dhcp.BuildDHCPOptions([]fields.TLVOption{
    dhcp.NewMessageTypeOption(dhcp.DHCPDISCOVER),
    dhcp.NewParamListOption([]uint8{
        dhcp.OptSubnetMask, dhcp.OptRouter, dhcp.OptDNS, dhcp.OptDomain,
    }),
    dhcp.EndOption(),
})

// 构建 Ethernet + IP + UDP + DHCP Discover
raw, err := goscapy.NewEthernet().
    SrcMAC("00:11:22:33:44:55").
    DstMAC("ff:ff:ff:ff:ff:ff").
    Over(goscapy.NewIP().
        SrcIP("0.0.0.0").
        DstIP("255.255.255.255")).
    Over(goscapy.NewUDP().
        SrcPort(68).
        DstPort(67)).
    Over(goscapy.NewDHCP().
        XID(0x12345678).
        CHAddr([]byte{0x00, 0x11, 0x22, 0x33, 0x44, 0x55}).
        Options(opts)).
    Build()

⚡ 快捷函数

go
// EtherIPUDPDHCP(srcMAC, dstMAC string, xid uint32, msgType uint8)
// 自动使用广播 IP 和 DHCP 客户端/服务器端口
raw, err := goscapy.EtherIPUDPDHCP(
    "00:11:22:33:44:55",   // 客户端 MAC
    "ff:ff:ff:ff:ff:ff",   // 广播
    0x12345678,            // 事务 ID
    dhcp.DHCPDISCOVER,      // 消息类型
)

🔬 解析

go
pkt, err := packet.Dissect(raw, packet.DissectEthernet)
fmt.Println(pkt.String()) // "Ethernet / IP / UDP / DHCP"

dhcpLayer := pkt.GetLayer("DHCP")

// 获取 DHCP 原始字段
xid, _ := dhcpLayer.Get("xid")
fmt.Printf("XID: 0x%08x\n", xid)

// 解析 DHCP 选项
optsData, _ := dhcpLayer.Get("options")
opts, _ := dhcp.ParseDHCPOptions(optsData.([]byte))
mt := dhcp.GetMessageType(opts)
fmt.Printf("MessageType=%d (%s)\n", mt, dhcp.MessageTypeString(mt))

DHCP Offer

构建 DHCP Offer — 服务器对 Discover 的响应。通过 YIAddr 提供 IP 地址，并包含租约选项。服务器将 Op 设为 BOOTREPLY，并在 SIAddr 中填入自己的地址。

🔨 Builder API

go
// 服务器提供的 DHCP 选项
opts := dhcp.BuildDHCPOptions([]fields.TLVOption{
    dhcp.NewMessageTypeOption(dhcp.DHCPOFFER),
    dhcp.NewServerIDOption("192.168.1.1"),
    dhcp.NewLeaseTimeOption(86400),            // 24 小时
    dhcp.NewSubnetMaskOption("255.255.255.0"),
    dhcp.NewRouterOption([]string{"192.168.1.1"}),
    dhcp.NewDNSOption([]string{"8.8.8.8", "8.8.4.4"}),
    dhcp.NewDomainOption("example.com"),
    dhcp.EndOption(),
})

raw, err := goscapy.NewEthernet().
    SrcMAC("00:aa:bb:cc:dd:ee").
    DstMAC("00:11:22:33:44:55").
    Over(goscapy.NewIP().
        SrcIP("192.168.1.1").
        DstIP("255.255.255.255")).
    Over(goscapy.NewUDP().
        SrcPort(67).
        DstPort(68)).
    Over(goscapy.NewDHCP().
        Op(dhcp.BOOTREPLY).
        XID(0x12345678).
        YIAddr("192.168.1.100").      // 分配的 IP
        SIAddr("192.168.1.1").        // 服务器 IP
        CHAddr([]byte{0x00, 0x11, 0x22, 0x33, 0x44, 0x55}).
        Options(opts)).
    Build()

🔬 解析

go
pkt, err := packet.Dissect(raw, packet.DissectEthernet)
dhcpLayer := pkt.GetLayer("DHCP")

// 读取 BOOTP 头部字段
yiaddr, _ := dhcpLayer.Get("yiaddr")
siaddr, _ := dhcpLayer.Get("siaddr")
fmt.Printf("分配 IP: %s  服务器: %s\n", yiaddr, siaddr)

// 解析 DHCP 选项
optsData, _ := dhcpLayer.Get("options")
opts, _ := dhcp.ParseDHCPOptions(optsData.([]byte))
mt := dhcp.GetMessageType(opts)
fmt.Printf("MessageType=%s\n", dhcp.MessageTypeString(mt))

// 查找特定选项
if lease := dhcp.GetDHCPOption(opts, dhcp.OptLeaseTime); lease != nil {
    fmt.Printf("租约时间: %d 秒\n",
        binary.BigEndian.Uint32(lease.Value))
}
注意：DHCP 选项使用 Type-Length-Value (TLV) 编码。BuildDHCPOptions() 将 TLVOption 结构体切片序列化为线格式。务必以 EndOption()（类型 255）结束选项列表。
