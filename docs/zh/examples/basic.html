Ethernet + IP + ICMP 回显请求

在以太网层构建 ICMP 回显请求（Echo Request）数据包，这是 ping 工具最常用的数据包类型。

🔨 Builder API

go
import "github.com/smallnest/goscapy/pkg/goscapy"

raw, err := goscapy.NewEthernet().
    SrcMAC("00:11:22:33:44:55").
    DstMAC("ff:ff:ff:ff:ff:ff").
    Over(goscapy.NewIP().
        SrcIP("10.0.0.1").
        DstIP("10.0.0.2")).
    Over(goscapy.NewICMP().
        Type(8).
        Code(0).
        ID(0x1234).
        Seq(1)).
    Build()

⚡ 快捷函数

go
// EtherIPICMP(dstMAC, dstIP string, icmpType, icmpCode uint8)
// 默认值：srcMAC="00:00:00:00:00:00"，srcIP="0.0.0.0"，id=0，seq=0
raw, err := goscapy.EtherIPICMP("ff:ff:ff:ff:ff:ff", "10.0.0.2", 8, 0)

🔬 解析

go
import "github.com/smallnest/goscapy/pkg/packet"

pkt, err := packet.Dissect(raw, packet.DissectEthernet)
fmt.Println(pkt.String()) // "Ethernet / IP / ICMP"

eth := packet.Ethernet(pkt)
fmt.Printf("SrcMAC: %s → DstMAC: %s\n", eth.SrcMAC(), eth.DstMAC())

ip := packet.IP(pkt)
fmt.Printf("%s → %s  TTL=%d\n", ip.SrcIP(), ip.DstIP(), ip.TTL())

📦 序列化十六进制 (42 字节)

hex
// Ethernet (14 字节)
ff ff ff ff ff ff 00 11 22 33 44 55 08 00
// IP (20 字节)
45 00 00 1c 00 00 00 00 40 01 66 df 0a 00 00 01 0a 00 00 02
// ICMP (8 字节)
08 00 e5 ca 12 34 00 01

Ethernet + IP + TCP SYN

构建 TCP SYN 数据包 — TCP 三次握手的第一步。

🔨 Builder API

go
import "github.com/smallnest/goscapy/pkg/layers"

raw, err := goscapy.NewEthernet().
    SrcMAC("00:11:22:33:44:55").
    DstMAC("ff:ff:ff:ff:ff:ff").
    Over(goscapy.NewIP().
        SrcIP("192.168.1.1").
        DstIP("8.8.8.8")).
    Over(goscapy.NewTCP().
        SrcPort(12345).
        DstPort(80).
        Flags(layers.TCPSyn).
        Seq(1000)).
    Build()

⚡ 快捷函数

go
// EtherIPTCP(srcMAC, dstMAC, srcIP, dstIP string, srcPort, dstPort uint16, flags uint8)
raw, err := goscapy.EtherIPTCP(
    "00:11:22:33:44:55", "ff:ff:ff:ff:ff:ff",
    "192.168.1.1", "8.8.8.8",
    12345, 80, layers.TCPSyn,
)

🔬 解析

go
pkt, err := packet.Dissect(raw, packet.DissectEthernet)
fmt.Println(pkt.String()) // "Ethernet / IP / TCP"

tcp := packet.TCP(pkt)
fmt.Printf("端口 %d → %d  Flags=0x%02x  Seq=%d\n",
    tcp.SrcPort(), tcp.DstPort(), tcp.Flags(), tcp.Seq())

📦 序列化十六进制 (54 字节)

hex
// Ethernet (14 字节)
ff ff ff ff ff ff 00 11 22 33 44 55 08 00
// IP (20 字节)
45 00 00 28 00 00 00 00 40 06 a9 12 c0 a8 01 01 08 08 08 08
// TCP (20 字节)
30 39 00 50 00 00 03 e8 00 00 00 00 50 02 20 00 45 e1 00 00

Ethernet + IP + UDP

构建 UDP 数据报 — 用于 DNS 查询、视频流等无连接协议。

🔨 Builder API

go
raw, err := goscapy.NewEthernet().
    SrcMAC("00:11:22:33:44:55").
    DstMAC("ff:ff:ff:ff:ff:ff").
    Over(goscapy.NewIP().
        SrcIP("192.168.1.1").
        DstIP("8.8.8.8")).
    Over(goscapy.NewUDP().
        SrcPort(12345).
        DstPort(53)).
    Build()

⚡ 快捷函数

go
// EtherIPUDP(srcMAC, dstMAC, srcIP, dstIP string, srcPort, dstPort uint16)
raw, err := goscapy.EtherIPUDP(
    "00:11:22:33:44:55", "ff:ff:ff:ff:ff:ff",
    "192.168.1.1", "8.8.8.8",
    12345, 53,
)

🔬 解析

go
pkt, err := packet.Dissect(raw, packet.DissectEthernet)
fmt.Println(pkt.String()) // "Ethernet / IP / UDP"

udp := packet.UDP(pkt)
fmt.Printf("端口 %d → %d  长度=%d\n",
    udp.SrcPort(), udp.DstPort(), udp.Len())

hex

Ethernet + ARP 请求

构建 ARP "who-has" 请求 — 用于在局域网中发现目标 IP 对应的 MAC 地址。

🔨 Builder API

go
raw, err := goscapy.NewEthernet().
    SrcMAC("aa:bb:cc:dd:ee:ff").
    DstMAC("ff:ff:ff:ff:ff:ff").
    Over(goscapy.NewARP().
        Op(layers.ARPWhoHas).
        SrcMAC("aa:bb:cc:dd:ee:ff").
        SrcIP("192.168.1.1").
        DstMAC("00:00:00:00:00:00").
        DstIP("192.168.1.100")).
    Build()

⚡ 快捷函数

go
// EtherARP(srcMAC, dstMAC, psrc, pdst string, op uint16)
raw, err := goscapy.EtherARP(
    "aa:bb:cc:dd:ee:ff", "ff:ff:ff:ff:ff:ff",
    "192.168.1.1", "192.168.1.100",
    layers.ARPWhoHas,
)

🔬 解析

go
pkt, err := packet.Dissect(raw, packet.DissectEthernet)
fmt.Println(pkt.String()) // "Ethernet / ARP"

arp := packet.ARP(pkt)
fmt.Printf("Op=%d  %s/%s → %s/%s\n",
    arp.Op(), arp.SrcMAC(), arp.SrcIP(), arp.DstMAC(), arp.DstIP())

📦 序列化十六进制 (42 字节)

hex
// Ethernet (14 字节)
ff ff ff ff ff ff aa bb cc dd ee ff 08 06
// ARP (28 字节)
00 01 08 00 06 04 00 01
aa bb cc dd ee ff c0 a8 01 01
00 00 00 00 00 00 c0 a8 01 64

IP + ICMP（无以太网头）

从 IP 层开始构建 ICMP 数据包 — 适用于 L3 层 Send() 场景，操作系统会自动添加以太网头。

🔨 Builder API

go
raw, err := goscapy.NewIP().
    SrcIP("10.0.0.1").
    DstIP("10.0.0.2").
    Over(goscapy.NewICMP().
        Type(8).
        Code(0)).
    Build()

⚡ 快捷函数

go
// IPICMP(srcIP, dstIP string, icmpType, icmpCode uint8)
raw, err := goscapy.IPICMP("10.0.0.1", "10.0.0.2", 8, 0)

🔬 解析

go
// 使用 DissectIP（无以太网头）
pkt, err := packet.Dissect(raw, packet.DissectIP)
fmt.Println(pkt.String()) // "IP / ICMP"

ip := packet.IP(pkt)
fmt.Printf("%s → %s  Proto=%d\n", ip.SrcIP(), ip.DstIP(), ip.Proto())

📦 序列化十六进制 (28 字节)

hex
// IP (20 字节)
45 00 00 1c 00 00 00 00 40 01 66 cd 0a 00 00 01 0a 00 00 02
// ICMP (8 字节)
08 00 f5 dc 00 00 00 00

IP + TCP（无以太网头）

从 IP 层开始构建 TCP SYN 数据包 — 配合 Send() 进行 L3 层发送。

🔨 Builder API

go
raw, err := goscapy.NewIP().
    SrcIP("192.168.1.1").
    DstIP("8.8.8.8").
    Over(goscapy.NewTCP().
        SrcPort(12345).
        DstPort(80).
        Flags(layers.TCPSyn).
        Seq(1000)).
    Build()

⚡ 快捷函数

go
// IPTCP(srcIP, dstIP string, srcPort, dstPort uint16, flags uint8)
raw, err := goscapy.IPTCP("192.168.1.1", "8.8.8.8", 12345, 80, layers.TCPSyn)

🔬 解析

go
pkt, err := packet.Dissect(raw, packet.DissectIP)
fmt.Println(pkt.String()) // "IP / TCP"

tcp := packet.TCP(pkt)
fmt.Printf("端口 %d → %d  SYN=%v\n",
    tcp.SrcPort(), tcp.DstPort(), tcp.Flags()&layers.TCPSyn != 0)

hex
// IP (20 字节)
45 00 00 28 00 00 00 00 40 06 a9 12 c0 a8 01 01 08 08 08 08
// TCP (20 字节)
