Ethernet + IP + GRE + 内层 IP

构建封装内层 IP 载荷的 GRE 隧道数据包。GRE 常用于站点间 VPN 和覆盖网络。设置 .Key()、.Seq() 或 .SetChecksum() 会自动启用对应的 GRE 标志位（K、S、C 位）。

🔨 Builder API

go
import (
    "github.com/smallnest/goscapy/pkg/goscapy"
    "github.com/smallnest/goscapy/pkg/layers/gre"
)

// 构建待封装的内层 IP 数据包
innerPayload, _ := goscapy.NewIP().
    SrcIP("10.1.0.1").
    DstIP("10.2.0.1").
    Over(goscapy.NewICMP().
        Type(8).Code(0)).
    Build()

// 外层 GRE 隧道：Ethernet + IP + GRE + 内层 IP
raw, err := goscapy.NewEthernet().
    SrcMAC("00:11:22:33:44:55").
    DstMAC("66:77:88:99:aa:bb").
    Over(goscapy.NewIP().
        SrcIP("192.168.1.1").
        DstIP("192.168.2.1")).
    Over(goscapy.NewGRE().
        ProtocolType(gre.ProtoIP).       // 0x0800 = 封装 IP
        Key(0x00001000).                   // 自动设置 K 标志
        Seq(42).                          // 自动设置 S 标志
        Payload(innerPayload)).            // 内层 IP 数据包原始字节
    Build()

⚡ 快捷函数

go
// EtherIPGRE(srcMAC, dstMAC, srcIP, dstIP string, protoType uint16, key uint32, innerPayload []byte)
// key=0 表示不包含 Key 字段；非零值自动设置 K 标志
raw, err := goscapy.EtherIPGRE(
    "00:11:22:33:44:55", "66:77:88:99:aa:bb",
    "192.168.1.1", "192.168.2.1",
    gre.ProtoIP,       // ProtocolType = 0x0800 (IP)
    0x00001000,        // Key 值
    innerPayload,      // 内层 IP + ICMP 字节
)

🔬 解析

go
import "github.com/smallnest/goscapy/pkg/packet"

pkt, err := packet.Dissect(raw, packet.DissectEthernet)
fmt.Println(pkt.String()) // "Ethernet / IP / GRE"

// 访问 GRE 字段
greLayer := packet.Layer(pkt, "GRE")
fmt.Printf("ProtoType=0x%04x  HasKey=%v  HasSeq=%v\n",
    gre.GetProtocolType(greLayer),
    gre.HasKey(greLayer),
    gre.HasSeq(greLayer))
if gre.HasKey(greLayer) {
    fmt.Printf("Key=0x%08x\n", gre.GetKey(greLayer))
}
if gre.HasSeq(greLayer) {
    fmt.Printf("Seq=%d\n", gre.GetSeq(greLayer))
}
注意：GRE 条件字段（Key、Seq、Checksum）仅在显式设置时才包含在序列化输出中。通过 Builder 设置字段时会自动启用 GRE 头中对应的标志位。

GRE 带校验和字段

GRE 支持可选的校验和字段用于完整性验证。设置 .SetChecksum() 会启用 C 标志并在头部中包含校验和字段。使用 .Build() 时，校验和值可以设为 0 — 序列化时会自动计算。

🔨 Builder API

go
// GRE 包含所有可选字段：Checksum + Key + Sequence
raw, err := goscapy.NewEthernet().
    SrcMAC("00:11:22:33:44:55").
    DstMAC("66:77:88:99:aa:bb").
    Over(goscapy.NewIP().
        SrcIP("192.168.1.1").
        DstIP("192.168.2.1")).
    Over(goscapy.NewGRE().
        ProtocolType(gre.ProtoEthernet).  // 0x6558 = 封装 Ethernet
        SetChecksum(0).                   // 自动计算；启用 C 标志
        Key(0xABCD0001).
        Seq(100).
        Payload(innerEthernetFrame)).
    Build()

📦 GRE 头部布局

text
// GRE 头部（C+K+S 标志设置时，变长）
//
//   Flags (2 字节): C=1, R=0, K=1, S=1
//   ProtocolType (2 字节): 0x6558 (Ethernet) 或 0x0800 (IP)
//   Checksum + Reserved1 (4 字节) — 仅当 C=1
//   Key (4 字节) — 仅当 K=1
//   Sequence (4 字节) — 仅当 S=1
//
// 最小 GRE 头部 = 4 字节 (flags + protocol type)
// 包含所有可选字段 = 16 字节

Ethernet + IP + UDP + VXLAN + 内层 Ethernet

构建 VXLAN 隧道数据包 — 数据中心网络虚拟化的事实标准。VXLAN 将二层以太网帧封装在 UDP 数据报中，通过 24 位 VNI（VXLAN 网络标识符）进行区分。

🔨 Builder API

go
// 待封装的内层以太网帧
innerFrame, _ := goscapy.NewEthernet().
    SrcMAC("aa:bb:cc:dd:ee:01").
    DstMAC("aa:bb:cc:dd:ee:02").
    Over(goscapy.NewIP().
        SrcIP("10.1.0.1").
        DstIP("10.1.0.2")).
    Over(goscapy.NewTCP().
        SrcPort(12345).DstPort(80)).
    Build()

// 外层 VXLAN 隧道
raw, err := goscapy.NewEthernet().
    SrcMAC("00:11:22:33:44:55").
    DstMAC("66:77:88:99:aa:bb").
    Over(goscapy.NewIP().
        SrcIP("192.168.1.1").
        DstIP("192.168.2.1")).
    Over(goscapy.NewUDP().
        SrcPort(4789).       // 标准 VXLAN UDP 端口
        DstPort(4789)).
    Over(goscapy.NewVXLAN().
        VNI(100500).             // 24 位 VXLAN 网络标识符
        Payload(innerFrame)).      // 内层以太网帧
    Build()

⚡ 快捷函数

go
// EtherIPUDPVXLAN(srcMAC, dstMAC, srcIP, dstIP string, vni uint32, innerPayload []byte)
// 默认使用 UDP 端口 4789 作为源端口和目标端口
raw, err := goscapy.EtherIPUDPVXLAN(
    "00:11:22:33:44:55", "66:77:88:99:aa:bb",
    "192.168.1.1", "192.168.2.1",
    100500,          // VNI
    innerFrame,      // 内层以太网帧字节
)

🔬 解析

go
import "github.com/smallnest/goscapy/pkg/layers/vxlan"

pkt, err := packet.Dissect(raw, packet.DissectEthernet)
fmt.Println(pkt.String()) // "Ethernet / IP / UDP / VXLAN"

// 访问 VXLAN 字段
vxlanLayer := packet.Layer(pkt, "VXLAN")
fmt.Printf("VNI=%d  Flags=0x%02x\n",
    vxlan.GetVNI(vxlanLayer),
    vxlan.GetFlags(vxlanLayer))
提示：VXLAN 的 VNI 是 24 位值（0–16,777,215）。I 标志（flags 字节的第 3 位）在有效 VXLAN 数据包中始终设置，由 Builder 自动处理。

Ethernet + Dot1Q (VLAN) + IP

构建 802.1Q VLAN 标记数据包。Dot1Q 在以太网头部和载荷之间插入 4 字节的 shim，携带 VLAN ID (VID)、优先级代码点 (PCP) 和丢弃合格指示符 (DEI)。

🔨 Builder API

go
import "github.com/smallnest/goscapy/pkg/layers/dot1q"

raw, err := goscapy.NewEthernet().
    SrcMAC("00:11:22:33:44:55").
    DstMAC("ff:ff:ff:ff:ff:ff").
    Over(goscapy.NewDot1Q().
        VID(100).            // VLAN ID (0–4095)
        PCP(5).             // 优先级代码点 (0–7)
        DEI(false).          // 丢弃合格指示符
        Type(0x0800)).      // 内层 EtherType = IPv4
    Over(goscapy.NewIP().
        SrcIP("192.168.1.1").
        DstIP("10.0.0.1")).
    Over(goscapy.NewTCP().
        SrcPort(12345).DstPort(80)).
    Build()

⚡ 快捷函数

go
// EtherDot1QIP(srcMAC, dstMAC, srcIP, dstIP string, vid uint16)
// 默认：PCP=0, DEI=false, Type=0x0800 (IPv4), TPID=0x8100
raw, err := goscapy.EtherDot1QIP(
    "00:11:22:33:44:55", "ff:ff:ff:ff:ff:ff",
    "192.168.1.1", "10.0.0.1",
    100,  // VLAN ID
)

🔬 解析

go
pkt, err := packet.Dissect(raw, packet.DissectEthernet)
fmt.Println(pkt.String()) // "Ethernet / Dot1Q / IP / TCP"

// 访问 Dot1Q 字段
dot1qLayer := packet.Layer(pkt, "Dot1Q")
fmt.Printf("VID=%d  PCP=%d  DEI=%v  Type=0x%04x\n",
    dot1q.GetVID(dot1qLayer),
    dot1q.GetPCP(dot1qLayer),
    dot1q.GetDEI(dot1qLayer),
    dot1q.GetType(dot1qLayer))

📦 Dot1Q 头部布局 (4 字节)

text
// Dot1Q 4 字节标签插入在以太网 src/dst MAC 之后
//
//   TPID (2 字节): 0x8100 (802.1Q) 或 0x88A8 (802.1ad/QinQ)
//   TCI (2 字节):
//     PCP  (3 位) — 优先级代码点 (0–7)
//     DEI  (1 位) — 丢弃合格指示符
//     VID (12 位) — VLAN 标识符 (0–4095)

QinQ — 双层 VLAN 标签

QinQ (802.1ad) 堆叠两个 VLAN 标签 — 外层服务标签 (S-tag) 使用 TPID 0x88A8，内层客户标签 (C-tag) 使用 TPID 0x8100。这允许服务提供商透明地隧道传输客户 VLAN。

🔨 Builder API

go
raw, err := goscapy.NewEthernet().
    SrcMAC("00:11:22:33:44:55").
    DstMAC("ff:ff:ff:ff:ff:ff").
    Over(goscapy.NewDot1Q().           // 外层 S-tag (服务)
        TPID(dot1q.TPID8021AD).      // 0x88A8 = 802.1ad
        VID(2000).
        PCP(3).
        DEI(false)).
    Over(goscapy.NewDot1Q().           // 内层 C-tag (客户)
        TPID(dot1q.TPID8021Q).       // 0x8100 = 802.1Q
        VID(100).
        PCP(0).
        DEI(false)).
    Over(goscapy.NewIP().
        SrcIP("192.168.1.1").
        DstIP("10.0.0.1")).
    Over(goscapy.NewICMP().
        Type(8).Code(0)).
    Build()

🔬 解析

go
pkt, err := packet.Dissect(raw, packet.DissectEthernet)
fmt.Println(pkt.String()) // "Ethernet / Dot1Q / Dot1Q / IP / ICMP"

// 访问外层 S-tag
outerTag := packet.Layer(pkt, "Dot1Q")
fmt.Printf("S-Tag: VID=%d  PCP=%d\n",
    dot1q.GetVID(outerTag), dot1q.GetPCP(outerTag))

// 访问内层 C-tag（第二个 Dot1Q 层）
dot1qLayers := packet.Layers(pkt, "Dot1Q")
if len(dot1qLayers) > 1 {
    innerTag := dot1qLayers[1]
    fmt.Printf("C-Tag: VID=%d  PCP=%d\n",
        dot1q.GetVID(innerTag), dot1q.GetPCP(innerTag))
}
注意：外层 Dot1Q 使用 TPID8021AD（0x88A8）以区别于单标签帧。内层标签使用标准 TPID8021Q（0x8100）。解析引擎会自动识别嵌套的 Dot1Q 层。
