IPv6 + ICMPv6 Echo Request

Build an IPv6 ping packet. ICMPv6 Echo uses a two-layer structure: the ICMPv6 base header (Type + Code + Checksum) and the Echo sub-layer (ID + Seq). IPv6 fields TC (Traffic Class), FL (Flow Label), and HLim (Hop Limit) control QoS and routing behavior. The pseudo-header checksum is computed automatically.

🔨 Builder API

go
import (
    "github.com/smallnest/goscapy/pkg/goscapy"
    "github.com/smallnest/goscapy/pkg/layers"
)

raw, err := goscapy.NewIPv6().
    SrcIP("fd00::1").
    DstIP("fd00::2").
    TC(0).                // Traffic Class (0–255)
    FL(0).                // Flow Label (20-bit, 0–1048575)
    HLim(64).              // Hop Limit
    Over(goscapy.NewICMPv6().
        Type(128).
        Code(0)).
    Over(&goscapy.RawBuilder{Layer: layers.NewICMPv6Echo(0x1234, 1)}).
    Build()

⚡ Shortcut

go
// IPv6ICMPv6Echo(srcIP, dstIP string, id, seq uint16)
raw, err := goscapy.IPv6ICMPv6Echo("fd00::1", "fd00::2", 0x1234, 1)

🔬 Dissect

go
pkt, err := packet.Dissect(raw, packet.DissectIPv6)
fmt.Println(pkt.String()) // "IPv6 / ICMPv6 / ICMPv6 Echo"

ip6 := pkt.GetLayer("IPv6")
srcIP, _ := ip6.Get("srcIP")
dstIP, _ := ip6.Get("dstIP")
hlim, _  := ip6.Get("hopLimit")
fmt.Printf("%s → %s  HLim=%d\n", srcIP, dstIP, hlim)

IPv6 + TCP SYN

Build a TCP SYN over IPv6. The Next Header field (NH) is set automatically based on the upper layer (6 = TCP). IPv6 pseudo-header checksum is used for TCP checksum computation.

🔨 Builder API

go
raw, err := goscapy.NewIPv6().
    SrcIP("fd00::1").
    DstIP("fd00::2").
    HLim(64).
    Over(goscapy.NewTCP().
        SrcPort(12345).
        DstPort(80).
        Flags(layers.TCPSyn).
        Seq(1000)).
    Build()

🔬 Dissect

go
pkt, err := packet.Dissect(raw, packet.DissectIPv6)
fmt.Println(pkt.String()) // "IPv6 / TCP"

tcp := pkt.GetLayer("TCP")
srcPort, _ := tcp.Get("srcPort")
dstPort, _ := tcp.Get("dstPort")
fmt.Printf("Port %d → %d  SYN\n", srcPort, dstPort)

IPv6 + UDP

Build a UDP datagram over IPv6. Next Header is set to 17 (UDP) automatically. This is the basis for IPv6-based DNS, DHCPv6, and VXLAN.

🔨 Builder API

go
raw, err := goscapy.NewIPv6().
    SrcIP("fd00::1").
    DstIP("fd00::2").
    Over(goscapy.NewUDP().
        SrcPort(12345).
        DstPort(53)).
    Build()

🔬 Dissect

pkt, err := packet.Dissect(raw, packet.DissectIPv6)

NDP Router Solicitation

Build an NDP Router Solicitation — used by an IPv6 host to discover routers on the local link. The message uses ICMPv6 type 133 and may include a Source Link-Layer Address option so the router can respond directly.

🔨 Builder API

go
import "github.com/smallnest/goscapy/pkg/layers/ndp"

// Build NDP options: Source Link-Layer Address
ndpOpts := ndp.BuildNDPOptions([]ndp.NDPOption{
    ndp.BuildSourceLinkLayerOption(net.HardwareAddr{0x00, 0x11, 0x22, 0x33, 0x44, 0x55}),
})

raw, err := goscapy.NewIPv6().
    SrcIP("fe80::211:22ff:fe33:4455").
    DstIP("ff02::2").            // all-routers multicast
    HLim(255).
    Over(goscapy.NewICMPv6().
        Type(ndp.NDPRouterSolicitation).  // 133
        Code(0)).
    Over(&goscapy.RawBuilder{Layer: layers.NewNDPRouterSolicitation()}).
    Build()

🔬 Dissect

go
pkt, err := packet.Dissect(raw, packet.DissectIPv6)
fmt.Println(pkt.String()) // "IPv6 / ICMPv6 / NDP Router Solicitation"

icmpLayer := pkt.GetLayer("ICMPv6")
msgType, _ := icmpLayer.Get("type")
fmt.Printf("ICMPv6 Type=%d (Router Solicitation)\n", msgType)

NDP Neighbor Advertisement

Build an NDP Neighbor Advertisement — used to respond to Neighbor Solicitations or announce a link-layer address change. Flags Router, Solicited, and Override control the recipient's neighbor cache behavior.

🔨 Builder API

go
// Build NDP options: Target Link-Layer Address
ndpOpts := ndp.BuildNDPOptions([]ndp.NDPOption{
    ndp.BuildTargetLinkLayerOption(net.HardwareAddr{0x00, 0xaa, 0xbb, 0xcc, 0xdd, 0xee}),
})

raw, err := goscapy.NewIPv6().
    SrcIP("fd00::1").
    DstIP("fd00::2").
    HLim(255).
    Over(goscapy.NewICMPv6().
        Type(ndp.NDPNeighborAdvertisement). // 136
        Code(0)).
    Over(&goscapy.RawBuilder{Layer: layers.NewNDPNeighborAdvertisement()}).
    Build()

🔬 Dissect

go
pkt, err := packet.Dissect(raw, packet.DissectIPv6)
fmt.Println(pkt.String()) // "IPv6 / ICMPv6 / NDP Neighbor Advertisement"

icmpLayer := pkt.GetLayer("ICMPv6")
msgType, _ := icmpLayer.Get("type")
fmt.Printf("ICMPv6 Type=%d (Neighbor Advertisement)\n", msgType)

ndpLayer := pkt.GetLayer("NDP Neighbor Advertisement")
if ndpLayer != nil {
    flags, _ := ndpLayer.Get("flags")
    fmt.Printf("Flags=0x%02x  Router=%v  Solicited=%v\n",
        flags,
        flags.(uint8)&ndp.NDPNARouter != 0,
        flags.(uint8)&ndp.NDPNASolicited != 0)
}

IPv6 Extension Header Chain

Build an IPv6 packet with a chain of extension headers: Hop-by-Hop Options → Fragment → TCP payload. Extension headers are stacked using Over() and the Next Header field is set automatically at each hop. The Fragment header carries offset and More Fragments (M) flag for IPv6 fragmentation.

🔨 Builder API

go
import "github.com/smallnest/goscapy/pkg/layers/ipv6"

// IPv6 with Hop-by-Hop → Fragment → TCP extension header chain
raw, err := goscapy.NewIPv6().
    SrcIP("fd00::1").
    DstIP("fd00::2").
    HLim(64).
    // Next Header = 0 (Hop-by-Hop) set automatically
    Over(&goscapy.RawBuilder{Layer: ipv6.NewIPv6HopByHop()}).
    // Next Header = 44 (Fragment) set automatically
    Over(&goscapy.RawBuilder{Layer: ipv6.NewIPv6Fragment()}).
    // Final Next Header = 6 (TCP)
    Over(goscapy.NewTCP().
        SrcPort(12345).
        DstPort(80).
        Flags(layers.TCPSyn)).
    Build()

🔬 Dissect

go
pkt, err := packet.Dissect(raw, packet.DissectIPv6)
fmt.Println(pkt.String())
// "IPv6 / IPv6 Hop-by-Hop / IPv6 Fragment / TCP"

// Walk the extension header chain
for _, layer := range pkt.Layers() {
    fmt.Printf("Layer: %s\n", layer.Proto())
}

// Access Fragment header fields
fragLayer := pkt.GetLayer("IPv6 Fragment")
if fragLayer != nil {
    fragData, _ := fragLayer.Get("fragment")
    offset := ipv6.IPv6FragmentOffset(fragData.(uint16))
    more    := ipv6.IPv6FragmentMore(fragData.(uint16))
    fmt.Printf("Fragment: offset=%d  more=%v\n", offset, more)
}
Note: IPv6 extension headers are chained via the Next Header field. Each extension header indicates the type of the following header. The final extension header points to the upper-layer protocol (TCP/UDP/ICMPv6). The builder sets Next Header automatically based on the Over() chain.
