DNS Query — A Record Lookup

Build a DNS query for an A record. The standard DNS query is encapsulated in Ethernet + IP + UDP port 53. The DNSQuestion struct holds the domain name, query type, and class. EncodeName() converts dotted notation to DNS wire-format labels.

🔨 Builder API

go
import (
    "github.com/smallnest/goscapy/pkg/goscapy"
    "github.com/smallnest/goscapy/pkg/layers/dns"
)

// Define the DNS question
questions := []dns.DNSQuestion{
    {Name: "example.com", Type: dns.QtypeA, Class: dns.QclassIN},
}

// Build Ethernet + IP + UDP + DNS query
raw, err := goscapy.NewEthernet().
    SrcMAC("00:11:22:33:44:55").
    DstMAC("ff:ff:ff:ff:ff:ff").
    Over(goscapy.NewIP().
        SrcIP("192.168.1.1").
        DstIP("8.8.8.8")).
    Over(goscapy.NewUDP().
        SrcPort(12345).
        DstPort(53)).
    Over(goscapy.NewDNS().
        ID(0x1234).
        Flags(0x0100).         // RD=1 (recursion desired)
        Questions(questions)).
    Build()

⚡ Shortcut

go
// EtherIPUDPDNS(srcMAC, dstMAC, srcIP, dstIP string, dnsPort uint16, questions []dns.DNSQuestion)
questions := []dns.DNSQuestion{
    {Name: "example.com", Type: dns.QtypeA, Class: dns.QclassIN},
}
raw, err := goscapy.EtherIPUDPDNS(
    "00:11:22:33:44:55", "ff:ff:ff:ff:ff:ff",
    "192.168.1.1", "8.8.8.8",
    53,
    questions,
)

🔬 Dissect

go
import "github.com/smallnest/goscapy/pkg/packet"

pkt, err := packet.Dissect(raw, packet.DissectEthernet)
fmt.Println(pkt.String()) // "Ethernet / IP / UDP / DNS"

// Extract DNS question section
dnsLayer := pkt.GetLayer("DNS")
questions, err := dns.GetQuestions(dnsLayer)
for _, q := range questions {
    fmt.Printf("Query: %s  Type=%d  Class=%d\n", q.Name, q.Type, q.Class)
}

DNS Response — A Record Answer

Build a DNS response with an A record answer. The response has the QR flag set (flags 0x8180 = QR + RD + RA) and includes both a question section and an answer section. Use BuildARData() to construct the RDATA for A records.

🔨 Builder API

go
questions := []dns.DNSQuestion{
    {Name: "example.com", Type: dns.QtypeA, Class: dns.QclassIN},
}
answers := []dns.DNSRR{
    {
        Name:     "example.com",
        Type:     dns.QtypeA,
        Class:    dns.QclassIN,
        TTL:      300,
        RDLength: 4,
        RData:    dns.BuildARData("93.184.216.34"),
    },
}

// Build DNS response data with questions + answers
dnsData := dns.BuildDNSMessage(questions, answers, nil, nil)

raw, err := goscapy.NewEthernet().
    SrcMAC("00:aa:bb:cc:dd:ee").
    DstMAC("00:11:22:33:44:55").
    Over(goscapy.NewIP().
        SrcIP("8.8.8.8").
        DstIP("192.168.1.1")).
    Over(goscapy.NewUDP().
        SrcPort(53).
        DstPort(12345)).
    Over(goscapy.NewDNS().
        ID(0x1234).
        Flags(0x8180).        // QR=1, RD=1, RA=1
        Data(dnsData)).
    Build()

🔬 Dissect

go
pkt, err := packet.Dissect(raw, packet.DissectEthernet)

dnsLayer := pkt.GetLayer("DNS")

// Parse questions
questions, _ := dns.GetQuestions(dnsLayer)

// Parse answer resource records
answers, _ := dns.GetAnswers(dnsLayer)
for _, rr := range answers {
    fmt.Printf("Answer: %s  Type=%d  TTL=%d", rr.Name, rr.Type, rr.TTL)
    if rr.Type == dns.QtypeA {
        fmt.Printf("  IP=%s", dns.ParseARData(rr.RData))
    }
    fmt.Println()
}
Tip: DNS name encoding uses label compression — repeated domain names in the response reference earlier occurrences via pointers (top 2 bits = 0xC0). DecodeName() handles this automatically.

DHCP Discover

Build a DHCP Discover packet — the first message a client sends to find a DHCP server on the network. It uses broadcast addressing (255.255.255.255) and UDP ports 68 (client) → 67 (server). The MessageType TLV option (53) is set to DHCPDISCOVER.

🔨 Builder API

go
import (
    "github.com/smallnest/goscapy/pkg/goscapy"
    "github.com/smallnest/goscapy/pkg/layers/dhcp"
)

// Build DHCP options: MessageType + Parameter Request List + End
opts := dhcp.BuildDHCPOptions([]fields.TLVOption{
    dhcp.NewMessageTypeOption(dhcp.DHCPDISCOVER),
    dhcp.NewParamListOption([]uint8{
        dhcp.OptSubnetMask, dhcp.OptRouter, dhcp.OptDNS, dhcp.OptDomain,
    }),
    dhcp.EndOption(),
})

// Build Ethernet + IP + UDP + DHCP Discover
raw, err := goscapy.NewEthernet().
    SrcMAC("00:11:22:33:44:55").
    DstMAC("ff:ff:ff:ff:ff:ff").
    Over(goscapy.NewIP().
        SrcIP("0.0.0.0").
        DstIP("255.255.255.255")).
    Over(goscapy.NewUDP().
        SrcPort(68).
        DstPort(67)).
    Over(goscapy.NewDHCP().
        XID(0x12345678).
        CHAddr([]byte{0x00, 0x11, 0x22, 0x33, 0x44, 0x55}).
        Options(opts)).
    Build()

⚡ Shortcut

go
// EtherIPUDPDHCP(srcMAC, dstMAC string, xid uint32, msgType uint8)
// Uses broadcast IP and DHCP client/server ports automatically
raw, err := goscapy.EtherIPUDPDHCP(
    "00:11:22:33:44:55",   // client MAC
    "ff:ff:ff:ff:ff:ff",   // broadcast
    0x12345678,            // transaction ID
    dhcp.DHCPDISCOVER,      // message type
)

🔬 Dissect

go
pkt, err := packet.Dissect(raw, packet.DissectEthernet)
fmt.Println(pkt.String()) // "Ethernet / IP / UDP / DHCP"

dhcpLayer := pkt.GetLayer("DHCP")

// Get raw DHCP fields
xid, _ := dhcpLayer.Get("xid")
fmt.Printf("XID: 0x%08x\n", xid)

// Parse DHCP options from the options field
optsData, _ := dhcpLayer.Get("options")
opts, _ := dhcp.ParseDHCPOptions(optsData.([]byte))
mt := dhcp.GetMessageType(opts)
fmt.Printf("MessageType=%d (%s)\n", mt, dhcp.MessageTypeString(mt))

DHCP Offer

Build a DHCP Offer — the server's response to a Discover. It proposes an IP address via YIAddr and includes lease options. The server sets Op to BOOTREPLY and populates SIAddr with its own address.

🔨 Builder API

go
// Server-offered DHCP options
opts := dhcp.BuildDHCPOptions([]fields.TLVOption{
    dhcp.NewMessageTypeOption(dhcp.DHCPOFFER),
    dhcp.NewServerIDOption("192.168.1.1"),
    dhcp.NewLeaseTimeOption(86400),            // 24 hours
    dhcp.NewSubnetMaskOption("255.255.255.0"),
    dhcp.NewRouterOption([]string{"192.168.1.1"}),
    dhcp.NewDNSOption([]string{"8.8.8.8", "8.8.4.4"}),
    dhcp.NewDomainOption("example.com"),
    dhcp.EndOption(),
})

raw, err := goscapy.NewEthernet().
    SrcMAC("00:aa:bb:cc:dd:ee").
    DstMAC("00:11:22:33:44:55").
    Over(goscapy.NewIP().
        SrcIP("192.168.1.1").
        DstIP("255.255.255.255")).
    Over(goscapy.NewUDP().
        SrcPort(67).
        DstPort(68)).
    Over(goscapy.NewDHCP().
        Op(dhcp.BOOTREPLY).
        XID(0x12345678).
        YIAddr("192.168.1.100").      // offered IP
        SIAddr("192.168.1.1").        // server IP
        CHAddr([]byte{0x00, 0x11, 0x22, 0x33, 0x44, 0x55}).
        Options(opts)).
    Build()

🔬 Dissect

go
pkt, err := packet.Dissect(raw, packet.DissectEthernet)
dhcpLayer := pkt.GetLayer("DHCP")

// Read BOOTP header fields
yiaddr, _ := dhcpLayer.Get("yiaddr")
siaddr, _ := dhcpLayer.Get("siaddr")
fmt.Printf("Offered IP: %s  Server: %s\n", yiaddr, siaddr)

// Parse DHCP options
optsData, _ := dhcpLayer.Get("options")
opts, _ := dhcp.ParseDHCPOptions(optsData.([]byte))
mt := dhcp.GetMessageType(opts)
fmt.Printf("MessageType=%s\n", dhcp.MessageTypeString(mt))

// Look up specific options
if lease := dhcp.GetDHCPOption(opts, dhcp.OptLeaseTime); lease != nil {
    fmt.Printf("Lease Time: %d seconds\n",
        binary.BigEndian.Uint32(lease.Value))
}
Note: DHCP options use Type-Length-Value (TLV) encoding. BuildDHCPOptions() serializes a slice of TLVOption structs into wire format. Always end with EndOption() (type 255) to terminate the options list.
